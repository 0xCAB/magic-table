154 	function getRowHeight(row)
155 	{
156 		if (me.fisheyeEnabled) 
157 		{
158 			if (me.autoSize || tableModel.rowCount - tableModel.columnHeaderCount == 0) return 15;
159 			
160 			var h = (height - (fisheyeTopMargin + fisheyeBottomMargin)) / (tableModel.rowCount - tableModel.columnHeaderCount);
161 			return h - (h%1);
162 		}
163 		
164 		return tableModel.getRowHeight(row);
165 	}

167 	function getColumnWidth(column)
168 	{
169 		if (me.fisheyeEnabled) 
170 		{
171 			if (me.autoSize || tableModel.columnCount - tableModel.rowHeaderCount == 0) return 15;
172 			
173 			var w = (width - (fisheyeLeftMargin + fisheyeRightMargin)) / (tableModel.columnCount - tableModel.rowHeaderCount);
174 			return w - (w%1);
175 		}
176 		
177 		return tableModel.getColumnWidth(column);
178 	}

180 	function getSumOfRowHeights(justDataRows)
181 	{
182 		var sum = 0;
183 		var startRow;
184 		
185 		if (justDataRows == true)
186 			startRow = tableModel.columnHeaderCount;
187 		else
188 			startRow = 0;
189 			
190 		for (var i = startRow; i < tableModel.rowCount; i++)
191 			sum += tableModel.getRowHeight(i);
192 			
193 		return sum;
194 	}

205 			
206 		for (var i = startCol; i < tableModel.columnCount; i++)
207 			sum += tableModel.getColumnWidth(i);
208 			
209 		return sum;

223     function init()
224     {
225 		var t = me.fisheyeEnabled ? 1 : 0;
226 		contentColumnCount = (tableModel.columnCount - tableModel.rowHeaderCount * t);
227 		contentRowCount = (tableModel.rowCount - tableModel.columnHeaderCount * t);
228 		
229 		createTargetDiv();
230 		
231         fisheye = new greg.ross.visualisation.Fisheye();
232 		fisheye.setLensRadius(lensRadius);
233         var IE = document.all ? true : false
234         
235         if (!targetDiv) 
236             return;
237 			
238 		adjustTableMarginForCellDimensions();
239         setTableSize();
240         createTableCanvas(targetDiv, tableWidth, tableHeight);
241 		setTitle();
242 		createControlPanel();
243 		createControlPanelForm();
244         
245         if (!isCanvasSupported()) 
246             return;
247         
248         canvasContext = canvas.getContext("2d");
249 		greg.ross.visualisation.CanvasTextFunctions.enable(canvasContext);
250 		
251 		createBackBuffer(tableWidth, tableHeight);
252 		bufferContext = buffer.getContext("2d");
253 		greg.ross.visualisation.CanvasTextFunctions.enable(bufferContext);
254 		bufferContext.drawImage(canvas, 0, 0, tableWidth, tableHeight);
255 		
256 		bufferContext.clearRect(0, 0, canvas.width, canvas.height);
257 		bufferContext.fillStyle = backgroundColour;
258 		bufferContext.fillRect(0, 0, tableWidth, tableHeight);
259 		
260 		createCells();
261         renderAllCells(bufferContext);
262 		
263 		targetDiv.onmousemove = getBodyMouseXY;
264         addEvent(targetDiv, "mousemove", renderFocus);
265 		
266 		canvas.scroll;
267     }

431     function getBodyMouseXY(evt)
432     {
433         var x, y;
434        	
435         if (evt.pageX) 
436             x = evt.pageX;
437         else 
438             if (evt.clientX) 
439                 x = evt.clientX +
440                 (document.documentElement.scrollLeft ? document.documentElement.scrollLeft : document.body.scrollLeft);
441         
442         if (evt.pageY) 
443             y = evt.pageY;
444         else 
445             if (evt.clientY) 
446                 y = evt.clientY +
447                 (document.documentElement.scrollTop ? document.documentElement.scrollTop : document.body.scrollTop);
448         
449         bodyMousePosition = 
450         {
451             x: x,
452             y: y
453         };
454     }

456     var getElementPosition = function(element)
457     {
458         var posX = element.offsetLeft;
459         var posY = element.offsetTop;
460         try 
461         {
462             while (element.offsetParent) 
463             {
464                 posX += element.offsetParent.offsetLeft;
465                 posY += element.offsetParent.offsetTop;
466                 
467 				if (element == document.getElementsByTagName('body')[0]) 
468                     break
469                 else 
470                     element = element.offsetParent;
471             }
472         } 
473         catch (e) 
474         {
475         }
476 		
477         return {x:posX, y:posY};
478     }

502     function createCells()
503     {
504         var count = 0;
505         var leftX, rightX = 0, bottomY = fisheyeTopMargin + pixelOffset, topY;
506         var cellWidth;
507         var cellHeight;
508 		var newX;
509         
510         for (var i = 0; i < contentRowCount; i++) 
511         {
512 			cellHeight = getRowHeight(i);
513 			newX = fisheyeLeftMargin + pixelOffset;
514 			
515             for (var j = 0; j < contentColumnCount; j++) 
516             {
517 				cellWidth = getColumnWidth(j);
518                 leftX = newX;
519 				
520 				if (j == 0)
521 					topY = bottomY;
522 				
523 				rightX = leftX + cellWidth + pixelOffset;
524 				bottomY = topY + cellHeight;
525 				
526 				newX = rightX - pixelOffset;
527                 
528                 cells[count] = new greg.ross.visualisation.TableCell(leftX, topY, rightX, bottomY);
529                 count++;
530             }
531         }
532     }

542 	function renderScaleTextForFisheye(x, y, focusRow, focusColumn)
543 	{
544 		var contentWidth =  contentColumnCount * getColumnWidth();
545 		var contentLeftEdge = fisheyeLeftMargin + pixelOffset;
546 		var contentRightEdge = contentWidth + contentLeftEdge;
547 		
548 		var contentHeight = contentRowCount * getRowHeight();
549 		var contentTopEdge = fisheyeTopMargin + pixelOffset;
550 		var contentBottomEdge = contentHeight + contentTopEdge;
551 		var columnHeader = tableModel.getContentAt(0, focusColumn);
552 		var rowHeader = tableModel.getContentAt(focusRow, 0);
553 		
554 		if (x <= contentRightEdge + 20 && tableModel.columnHeaderCount > 0)
555 			renderAxisText(x, y, focusRow, focusColumn, contentWidth, contentLeftEdge, contentTopEdge,
556 				contentBottomEdge, columnHeader, "xAxis");
557 			
558 		if (y <= contentBottomEdge + 20 && tableModel.rowHeaderCount > 0)
559 			renderAxisText(y, x, focusRow, focusColumn, contentHeight, contentTopEdge, contentLeftEdge,
560 			contentRightEdge, rowHeader, "yAxis");
561 	}

563 	function renderAxisText(x, y, focusRow, focusColumn, contentWidth, contentLeftEdge, contentTopEdge,
564 			contentBottomEdge, cellText, axis)
565 	{
566 		if (cellText == null) return;
567 		
568 		var font = "sans";
569 	  	var fontsize = 16;
570 		
571 		x = contentWidth/2 + contentLeftEdge;
572 		
573 		if (y > contentTopEdge + lensRadius)
574 			y = DEFAULT_FISHEYE_MARGIN/2;
575 		else
576 			y = contentBottomEdge + (fisheyeBottomMargin/2) + 5;
577 		
578 		canvasContext.strokeStyle = 'rgb(255, 255, 255)';
579 		
580 		if (axis == "xAxis")
581 			canvasContext.drawTextCenter(font, fontsize, x, y, cellText + "");
582 		else
583 			canvasContext.drawTextCenter(font, fontsize, y, x, cellText + "", 90);
584 	}

586 	function renderFocus()
587 	{
588 		if (!me.fisheyeEnabled) return;
589 		
590 		var mousePosition = getRelativeMousePosition();
591 		var t = me.fisheyeEnabled ? 1 : 0;
592 		fisheye.setLensPosition(mousePosition.x, mousePosition.y);
593 		
594 		drawBackgroundCells();
595 		
596 		var r = lensRadius + Math.max(getRowHeight(), getColumnWidth()) + 10;
597 		var focusDiameter = r * 2;
598 		
599 		var marginOffsetX = fisheyeLeftMargin%getColumnWidth();
600 		var marginOffsetY = fisheyeTopMargin%getRowHeight();
601 		
602 		var mousePosX = mousePosition.x - r;
603 		var mousePosY = mousePosition.y - r;
604 		
605 		var horizontalOffset = (mousePosX - (mousePosX % getColumnWidth())) + marginOffsetX + pixelOffset;
606 		var verticalOffset = (mousePosY - (mousePosY % getRowHeight())) + marginOffsetY + pixelOffset;
607 		
608 		clearFocusArea(focusDiameter, marginOffsetX, marginOffsetY, mousePosX, mousePosY);
609 		
610 		var i = 0;
611 		var j = 0;
612 		var baseRow, baseCol;
613 		var cellValue;
614 		var maxCellArea = 0;
615 		var focusRow = null, focusColumn = null;
616 		
617 		var yPosition;
618 		do
619 		{
620 			xPosition = horizontalOffset;
621 			j = 0;
622 			do
623 			{
624 				yPosition = i * getRowHeight() + verticalOffset;
625 				xPosition = j * getColumnWidth() + horizontalOffset;
626 				
627 				baseCol = parseInt(xPosition/getColumnWidth()) - parseInt((fisheyeLeftMargin/getColumnWidth()));
628 				baseRow = parseInt(yPosition/getRowHeight()) - parseInt((fisheyeTopMargin/getRowHeight()));
629 				
630 				cellValue = null;
631 				
632 				if (baseRow < contentRowCount && baseCol < contentColumnCount)
633 					cellValue = tableModel.getContentAt(baseRow+tableModel.columnHeaderCount*t, baseCol+tableModel.rowHeaderCount*t);
634 				
635 				if (baseRow >= 0  && baseCol >= 0 && baseRow < tableModel.rowCount-tableModel.columnHeaderCount*t && baseCol < tableModel.columnCount-tableModel.rowHeaderCount*t) 
636 				{
637 					var cellCoordinates = getFisheyeCellCoordinates(xPosition, yPosition, getColumnWidth(), getRowHeight());
638 					
639 					if (cellCoordinates.x1 > 0 && cellCoordinates.y1 > 0) 
640 					{
641 						tableModel.getCellRendererAt(baseRow, baseCol, true).drawCell(canvasContext, baseCol+tableModel.rowHeaderCount, cellCoordinates, cellValue, me.barFillEnabled, getColumnWidth(), me.rowGradientEnabled);
642 						
643 						var cellArea = greg.ross.visualisation.TableGeometry.getArea(cellCoordinates.x1, cellCoordinates.y1, cellCoordinates.x2, cellCoordinates.y2, cellCoordinates.x3, cellCoordinates.y3, cellCoordinates.x4, cellCoordinates.y4);
644 						if (cellArea > maxCellArea) 
645 						{
646 							maxCellArea = cellArea;
647 							focusRow = baseRow;
648 							focusColumn = baseCol;
649 						}
650 					}
651 				}
652 				
653 				j++;
654 			}
655 			while (xPosition < (mousePosition.x + r) - getColumnWidth())
656 			i++;
657 		}
658 		while (yPosition < (mousePosition.y + r) - getRowHeight())
659 		
660 		if (focusRow != null && focusColumn != null)
661 			renderScaleTextForFisheye(mousePosition.x, mousePosition.y, focusRow+tableModel.columnHeaderCount, focusColumn+tableModel.rowHeaderCount);
662 	}

786 /**
787  * Return the area of the polygon defined by the given coordinates.
788  * @member greg.ross.visualisation.TableGeometry
789  * @param tlX top-left X-coordinate
790  * @param tlY top-left Y-coordinate
791  * @param trX top-right X-coordinate
792  * @param trY top-right Y-coordinate
793  * @param brX bottom-right X-coordinate
794  * @param brY bottom-right Y-coordinate
795  * @param blX bottom-left X-coordinate
796  * @param blY bottom-left Y-coordinate
797  */
798 greg.ross.visualisation.TableGeometry.getArea = function(tlX, tlY, trX, trY, brX, brY, blX, blY)
799 {
800 	return 0.5 * (tlX*trY - trX*tlY + trX*brY - brX*trY + brX*blY - blX*brY + blX*tlY - tlX*blY);
801 }

803 /**
804  * Return the center of the polygon specified by the given coordinates.
805  * @member greg.ross.visualisation.TableGeometry
806  * @param tlX top-left X-coordinate
807  * @param tlY top-left Y-coordinate
808  * @param trX top-right X-coordinate
809  * @param trY top-right Y-coordinate
810  * @param brX bottom-right X-coordinate
811  * @param brY bottom-right Y-coordinate
812  * @param blX bottom-left X-coordinate
813  * @param blY bottom-left Y-coordinate
814  */
815 greg.ross.visualisation.TableGeometry.getCentre = function(tlX, tlY, trX, trY, brX, brY, blX, blY)
816 {
817 	var second_factor;
818 	var polygon_area = greg.ross.visualisation.TableGeometry.getArea(tlX, tlY, trX, trY, brX, brY, blX, blY);
819 	
820     var X = 0;
821     var Y = 0;
822 	var points = [];
823 	points[0] = {x:tlX, y:tlY};
824 	points[1] = {x:trX, y:trY};
825 	points[2] = {x:brX, y:brY};
826 	points[3] = {x:blX, y:blY};
827 	points[4] = {x:tlX, y:tlY};
828     
829     for(var i = 0; i < 4; i++)
830     {
831         second_factor = points[i].x * points[i + 1].y - points[i + 1].x * points[i].y;
832         
833         X = (X + (points[i].x + points[i + 1].x) * second_factor);
834         Y = (Y + (points[i].y + points[i + 1].y) * second_factor);
835     }
836 
837     X /= (6 * polygon_area);
838     Y /= (6 * polygon_area);
839 
840     return {x:X, y:Y};
841 }

859 /**
860  * Return the amount by which a table cell must be filled to represent the numeric value it contains.
861  * @param tableModel the model containing the table data
862  * @param cellValue the value contained in the cell
863  * @param maxWidth the maximum allowable fill width. This is equal to the width of the cell
864  * @param column the cell's column index
865  */
866 greg.ross.visualisation.TableMath.calculateBarWidth = function(tableModel, cellValue, maxWidth, column)
867 {
868 	if (greg.ross.visualisation.TableMath.isNumber(cellValue)) 
869 	{
870 		var minValue = tableModel.getMinValueForColumn(column);
871 		var maxValue = tableModel.getMaxValueForColumn(column);
872 		
873 		if (minValue == maxValue) return maxWidth;
874 		
875 		var factor = (cellValue - minValue) / (maxValue - minValue);
876 		return factor * maxWidth;
877 	}
878 	return 0.5;
879 }

881 greg.ross.visualisation.TableMath.calculateBarWidthFraction = function(tableModel, cellValue, maxWidth, column)
882 {
883 	if (greg.ross.visualisation.TableMath.isNumber(cellValue)) 
884 	{
885 		var minValue = tableModel.getMinValueForColumn(column);
886 		var maxValue = tableModel.getMaxValueForColumn(column);
887 		
888 		if (minValue == maxValue) return 1;
889 		
890 		var factor = (cellValue - minValue) / (maxValue - minValue);
891 		return factor;
892 	}
893 	return 0;
894 }
