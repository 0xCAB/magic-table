 21 /**
 22  * This class renders cells when the fisheye feature is switched on.
 23  * @author Greg Ross
 24  * @constructor
 25  * @param tableModel
 26  * @param colourRamp the colour ramp from which the cell's back colour will be derived.
 27  */
 28 greg.ross.visualisation.FisheyeCellRenderer = function(tableModel, colourRamp)
 29 {
 30 	var colourGradient;
 31 	var defaultCellBackgroundColour = 'rgb(255, 255, 255)';
 32 	var gridColour = 'rgb(0, 0, 0)';
 33 	
 34 	function init()
 35 	{
 36 		colourGradient = greg.ross.visualisation.TableGradientColourProvider.createGradient(tableModel, colourRamp);
 37 	}
 38 	
 39 	function renderCellText(canvasContext, cellValue, tlX, tlY, trX, trY, brX, brY, blX, blY)
 40 	{
 41 		if (cellValue == null) return;
 42 		
 43 		var sValue = cellValue + "";
 44 		var font = "sans";
 45 	  	var fontsize = 10;
 46 		var centerPoint = greg.ross.visualisation.TableGeometry.getCentre(tlX, tlY, trX, trY, brX, brY, blX, blY);
 47 		canvasContext.drawTextCenter(font, fontsize, centerPoint.x, centerPoint.y, sValue);
 48 	}
 49 	
 50 	/**
 51 	 * The fisheye table calls this method to render the cell. Classes that extend a cell renderer
 52 	 * must implement this method.
 53 	 * @member greg.ross.visualisation.FisheyeCellRenderer
 54 	 * @param canvasContext the canvas context
 55 	 * @param column the column index
 56 	 * @param cellCoordinates the coordinates defining the cell poylgon
 57 	 * @param cellValue the value ot be rendered in the cell
 58 	 * @param barFillEnabled true if the bar-fill feature is turned on
 59 	 * @param cellWidth the width of the cell
 60 	 */
 61 	this.drawCell = function(canvasContext, column, cellCoordinates, cellValue, barFillEnabled, cellWidth, rowGradientEnabled)
 62 	{
 63 		var colr;
 64 		
 65 		if (rowGradientEnabled)
 66 			colr = tableModel.getColourGradientForRow(row);
 67 		else
 68 			colr = greg.ross.visualisation.TableGradientColourProvider.getColourFromValue(colourGradient, cellValue, defaultCellBackgroundColour);
 69 		
 70 		if (!barFillEnabled) 
 71 		{
 72 			canvasContext.fillStyle = colr;
 73 			canvasContext.beginPath();
 74 			canvasContext.moveTo(cellCoordinates.x1, cellCoordinates.y1);
 75 			canvasContext.lineTo(cellCoordinates.x2, cellCoordinates.y2);
 76 			canvasContext.lineTo(cellCoordinates.x3, cellCoordinates.y3);
 77 			canvasContext.lineTo(cellCoordinates.x4, cellCoordinates.y4);
 78 			canvasContext.lineTo(cellCoordinates.x1, cellCoordinates.y1);
 79 			canvasContext.fill();
 80 			
 81 			canvasContext.strokeStyle = gridColour;
 82 			canvasContext.lineWidth = 0.5;
 83 			canvasContext.stroke();
 84 		}
 85 		else
 86 		{
 87 			// Fill polygon background for fisheye cell.
 88 			canvasContext.fillStyle = defaultCellBackgroundColour;
 89 			canvasContext.beginPath();
 90 			canvasContext.moveTo(cellCoordinates.x1, cellCoordinates.y1);
 91 			canvasContext.lineTo(cellCoordinates.x2, cellCoordinates.y2);
 92 			canvasContext.lineTo(cellCoordinates.x3, cellCoordinates.y3);
 93 			canvasContext.lineTo(cellCoordinates.x4, cellCoordinates.y4);
 94 			canvasContext.lineTo(cellCoordinates.x1, cellCoordinates.y1);
 95 			canvasContext.fill();
 96 			
 97 			canvasContext.strokeStyle = gridColour;
 98 			canvasContext.lineWidth = 0.5;
 99 			canvasContext.stroke();
100 			
101 			// Calculate and fill polygon according to cell value.
102 			var widthFraction = greg.ross.visualisation.TableMath.calculateBarWidthFraction(tableModel, cellValue, cellWidth, column);
103 			
104 			canvasContext.fillStyle = colr;
105 			canvasContext.beginPath();
106 			canvasContext.moveTo(cellCoordinates.x1, cellCoordinates.y1);
107 			
108 			var dx = parseFloat((cellCoordinates.x2 - cellCoordinates.x1) * widthFraction) + cellCoordinates.x1;
109 			var dy = parseFloat((cellCoordinates.y2 - cellCoordinates.y1) * widthFraction) +  cellCoordinates.y1;
110 			canvasContext.lineTo(dx, dy);
111 			
112 			dx =  parseFloat((cellCoordinates.x3 - cellCoordinates.x4) * widthFraction) + cellCoordinates.x4;
113 			dy = parseFloat((cellCoordinates.y3 - cellCoordinates.y4) * widthFraction) + cellCoordinates.y4;
114 			
115 			canvasContext.lineTo(dx, dy-0.5);
116 			canvasContext.lineTo(cellCoordinates.x4, cellCoordinates.y4-0.5);
117 			canvasContext.lineTo(cellCoordinates.x1, cellCoordinates.y1);
118 			canvasContext.fill();
119 			
120 			// Stroke the path.			
121 			canvasContext.beginPath();
122 			canvasContext.moveTo(cellCoordinates.x1, cellCoordinates.y1);
123 			var dx = parseFloat((cellCoordinates.x2 - cellCoordinates.x1) * widthFraction) + cellCoordinates.x1;
124 			var dy = parseFloat((cellCoordinates.y2 - cellCoordinates.y1) * widthFraction) +  cellCoordinates.y1;
125 			canvasContext.lineTo(dx, dy);
126 			
127 			dx =  parseFloat((cellCoordinates.x3 - cellCoordinates.x4) * widthFraction) + cellCoordinates.x4;
128 			dy = parseFloat((cellCoordinates.y3 - cellCoordinates.y4) * widthFraction) + cellCoordinates.y4;
129 			
130 			canvasContext.moveTo(dx, dy);
131 			canvasContext.lineTo(cellCoordinates.x4, cellCoordinates.y4);
132 			canvasContext.lineTo(cellCoordinates.x1, cellCoordinates.y1);
133 			
134 			canvasContext.strokeStyle = gridColour;
135 			canvasContext.lineWidth = 0.5;
136 			canvasContext.stroke();
137 		}
138 		
139 		var cellArea = greg.ross.visualisation.TableGeometry.getArea(cellCoordinates.x1, cellCoordinates.y1, cellCoordinates.x2, cellCoordinates.y2, cellCoordinates.x3, cellCoordinates.y3, cellCoordinates.x4, cellCoordinates.y4);
140 		
141 		if (cellArea > 1500) 
142 			renderCellText(canvasContext, cellValue, cellCoordinates.x1, cellCoordinates.y1, cellCoordinates.x2, cellCoordinates.y2, cellCoordinates.x3, cellCoordinates.y3, cellCoordinates.x4, cellCoordinates.y4);
143 	}
144 	
145 	init();
146 }
