 21 /**
 22  * The table model defines the data and meta-data for the table.
 23  * @author Greg Ross
 24  * @constructor
 25  * @param rowCount the number of rows in the table, including headers
 26  * @param columnCount the number of columns in the able, including headers
 27  * @param defaultRowHeight the default row height when the fisheye feature is turned off
 28  * @param defaultColumnWidth the default column width when the fisheye feature is turned off
 29  * @param rowHeaderCount the number of row headers
 30  * @param columnHeaderCount the number of column headers
 31  * @param clrRamp the colour ramp applied to the table to encode numbers
 32  */
 33 function TableModel(rowCount, columnCount, defaultRowHeight, defaultColumnWidth, rowHeaderCount, columnHeaderCount, clrRamp)
 34 {
 35 	this.matrix = new Matrix();
 36 	this.rowCount = rowCount;
 37 	this.columnCount = columnCount;
 38 	this.rowHeaderCount = rowHeaderCount;
 39 	this.columnHeaderCount = columnHeaderCount;
 40 	this.minValue = 0;
 41 	this.maxValue = 0;
 42 	
 43 	this.minRowValues = [];
 44 	this.maxRowValues = [];
 45 	this.minColumnValues = [];
 46 	this.maxColumnValues = [];
 47 	
 48 	this.rowHeights = [];
 49 	this.columnWidths = [];
 50 	
 51 	var minRowHeight = 5;
 52 	var minColumnWidth = 5;
 53 	
 54 	var colourRamp;
 55 	
 56 	this.fisheyeColumnWidth = 15;
 57 	this.fisheyeRowHeight = 15;
 58 	
 59 	this.fisheyeCellRenderer = new FisheyeCellRenderer(this, colourRamp);
 60 	this.defaultCellRenderer = new DefaultCellRenderer(this, CellAlignment.CENTRE, colourRamp);
 61 	this.scaleCellRenderer = new ScaleCellRenderer(this, CellAlignment.CENTRE);
 62 	init();
 63 	
 64 	function init()
 65 	{
 66 		if (clrRamp)
 67 			colourRamp = clrRamp;
 68 		else
 69 			colourRamp = getDefaultColourRamp();
 70 	}
 71 	
 72 	function getDefaultColourRamp()
 73 	{
 74 		var colour1 = {red:0, green:0, blue:255};
 75 		var colour2 = {red:0, green:255, blue:255};
 76 		var colour3 = {red:0, green:255, blue:0};
 77 		var colour4 = {red:255, green:255, blue:0};
 78 		var colour5 = {red:255, green:0, blue:0};
 79 		return [colour1, colour2, colour3, colour4, colour5];
 80 	}
 81 	
 82 	/**
 83 	 * Return the cell render for the cell specified by the given row and column indices.
 84 	 * @member TableModel
 85 	 * @param row the row of the cell that is to be rendered
 86 	 * @param column the column of the cell that is to be rendered
 87 	 * @param isFisheyeCell true if the fisheye feature is turned on
 88 	 */
 89 	this.getCellRendererAt = function(row, column, isFisheyeCell)
 90 	{
 91 		if (isFisheyeCell) 
 92 			return this.fisheyeCellRenderer;
 93 		else 
 94 		{
 95 			if (TableMath.isNumber(this.rowHeaderCount))
 96 			{
 97 				if (column < this.rowHeaderCount)
 98 					return this.scaleCellRenderer;
 99 			}
100 			
101 			if (TableMath.isNumber(this.columnHeaderCount))
102 			{
103 				if (row < this.columnHeaderCount)
104 					return this.scaleCellRenderer;
105 			}
106 			
107 			return this.defaultCellRenderer;
108 		}
109 	}
110 	
111 	/**
112 	 * Return the height of the given row.
113 	 * @member TableModel
114 	 * @param row
115 	 */
116 	this.getRowHeight = function(row)
117 	{
118 		if (!TableMath.isNumber(row)) return defaultRowHeight;
119 		if (!this.rowHeights[row]) return defaultRowHeight;
120 		
121 		return this.rowHeights[row];
122 	}
123 	
124 	/**
125 	 * Return the width of the given column.
126 	 * @member TableModel
127 	 * @param column
128 	 */
129 	this.getColumnWidth = function(column)
130 	{
131 		if (!TableMath.isNumber(column)) return defaultColumnWidth;
132 		if (!this.columnWidths[column]) return defaultColumnWidth;
133 		
134 		return this.columnWidths[column];
135 	}
136 	
137 	/**
138 	 * Use this method to set the height of the given row.
139 	 * @member TableModel
140 	 * @param row
141 	 * @param height
142 	 */
143 	this.setRowHeight = function(row, height)
144 	{
145 		this.rowHeights[row] = height;
146 	}
147 	
148 	/**
149 	 * Use this method to set the width of the given column.
150 	 * @member TableModel
151 	 * @param column
152 	 * @param width
153 	 */
154 	this.setColumnWidth = function(column, width)
155 	{
156 		this.columnWidths[column] = width;
157 	}
158 	
159 	/**
160 	 * Call this method after adding numerical data to the table. This allows the table to
161 	 * track the maximum and minimum values across rows and columns.
162 	 * @member TableModel
163 	 */
164 	this.recalculateMinMaxValues = function()
165 	{
166 		var i = this.rowCount;
167 		var j = this.columnCount;
168 		
169 		this.minRowValues = [];
170 		this.maxRowValues = [];
171 		this.minColumnValues = [];
172 		this.maxColumnValues = [];
173 		
174 		do
175 		{
176 			j = this.columnCount;
177 			do
178 			{
179 				var row = this.rowCount - i;
180 				var column = this.columnCount - j
181 				var value = this.getContentAt(row, column);
182 				
183 				if (TableMath.isNumber(value)) 
184 				{
185 					if (this.minRowValues[row] == undefined) this.minRowValues[row] = value;
186 					if (this.maxRowValues[row] == undefined) this.maxRowValues[row] = value;
187 					if (this.minColumnValues[column] == undefined) this.minColumnValues[column] = value;
188 					if (this.maxColumnValues[column] == undefined) this.maxColumnValues[column] = value;
189 		
190 					if (value < this.minRowValues[row]) 
191 						this.minRowValues[row] = value;
192 					if (value > this.maxRowValues[row]) 
193 						this.maxRowValues[row] = value;
194 					if (value < this.minColumnValues[column]) 
195 						this.minColumnValues[column] = value;
196 					if (value > this.maxColumnValues[column]) 
197 						this.maxColumnValues[column] = value;
198 				}
199 			}	
200 			while(--j > 0)
201 		}
202 		while(--i > 0)
203 	}
204 	
205 	/**
206 	 * Return the value at the cell specified by the row and column parameters.
207 	 * @member TableModel
208 	 * @param row
209 	 * @param column
210 	 */
211 	this.getContentAt = function(row, column)
212 	{
213 		if (!isRowValid(row) || !isColumnValid(column)) return null;
214 		
215 		return this.matrix.get([row, column]);
216 	}
217 	
218 	/**
219 	 * Put a value in the table at the specified row and column.
220 	 * @member TableModel
221 	 * @param row
222 	 * @param column
223 	 * @param value
224 	 */
225 	this.setContentAt = function(row, column, value)
226 	{
227 		if (!isRowValid(row) || !isColumnValid(column)) return;
228 		
229 		this.matrix.put([row, column], value);
230 		this.fisheyeCellRenderer = new FisheyeCellRenderer(this, colourRamp);
231 		this.defaultCellRenderer = new DefaultCellRenderer(this, CellAlignment.CENTRE, colourRamp);
232 		
233 		if (TableMath.isNumber(value))
234 		{
235 			value = parseFloat(value);
236 			
237 			if (this.minRowValues[row] == undefined) this.minRowValues[row] = value;
238 			if (this.maxRowValues[row] == undefined) this.maxRowValues[row] = value;
239 			if (this.minColumnValues[column] == undefined) this.minColumnValues[column] = value;
240 			if (this.maxColumnValues[column] == undefined) this.maxColumnValues[column] = value;
241 			
242 			if (value < this.minRowValues[row]) this.minRowValues[row] = value;
243 			if (value > this.maxRowValues[row]) this.maxRowValues[row] = value;
244 			if (value < this.minColumnValues[column]) this.minColumnValues[column] = value;
245 			if (value > this.maxColumnValues[column]) this.maxColumnValues[column] = value;
246 		}
247 	}
248 	
249 	function isRowValid(row)
250 	{
251 		return row >= 0 && row < rowCount;
252 	}
253 	
254 	function isColumnValid(column)
255 	{
256 		return column >= 0 && column < columnCount;
257 	}
258 }
259 
260 /**
261  * Determine the smallest numeric value in the table.
262  * @member TableModel
263  */
264 TableModel.prototype.getMinValue = function()
265 {
266 	return this.matrix.minValue;
267 }
268 
269 /**
270  * Determine the largest numeric value in the table.
271  * @member TableModel
272  */
273 TableModel.prototype.getMaxValue = function()
274 {
275 	return this.matrix.maxValue;
276 }
277 
278 /**
279  * Return the smallest numeric value in the given row.
280  * @member TableModel
281  * @param row
282  */
283 TableModel.prototype.getMinValueForRow = function(row)
284 {
285 	return this.minRowValues[row];
286 }
287 
288 /**
289  * Return the largest numeric value in the given row.
290  * @member TableModel
291  * @param row
292  */
293 TableModel.prototype.getMaxValueForRow = function(row)
294 {
295 	return this.maxRowValues[row];
296 }
297 
298 /**
299  * Return the smallest numeric value in the given column.
300  * @member TableModel
301  * @param column
302  */
303 TableModel.prototype.getMinValueForColumn = function(column)
304 {
305 	return this.minColumnValues[column];
306 }
307 
308 /**
309  * Return the largest numeric value in the given column.
310  * @member TableModel
311  * @param column
312  */
313 TableModel.prototype.getMaxValueForColumn = function(column)
314 {
315 	return this.maxColumnValues[column];
316 }
