  2 function registerNameSpace(ns)
  3 {
  4     var nsParts = ns.split(".");
  5     var root = window;
  6     
  7     for (var i = 0; i < nsParts.length; i++) 
  8     {
  9         if (typeof root[nsParts[i]] == "undefined") 
 10             root[nsParts[i]] = new Object();
 11         
 12         root = root[nsParts[i]];
 13     }
 14 }

 54 greg.ross.visualisation.MagicTable.prototype.draw = function(data, options)
 55 {
 56     var defaultRowHeight = options.defaultRowHeight;
 57     var defaultColumnWidth = options.defaultColumnWidth;
 58     var tablePositionX = options.tablePositionX;
 59     var tablePositionY = options.tablePositionY;
 60     var tableHeight = options.tableHeight;
 61     var tableWidth = options.tableWidth;
 62     var rowHeaderCount = options.rowHeaderCount;
 63     var columnHeaderCount = options.columnHeaderCount;
 64     var rows = data.getNumberOfRows();
 65     var columns = data.getNumberOfColumns();
 66     var tableModel
 67     
 68     tableModel = new greg.ross.visualisation.TableModel(rows, columns, defaultRowHeight, defaultColumnWidth, rowHeaderCount, columnHeaderCount);
 69     
 70     var i = rows - 1;
 71     var j;
 72     
 73     do 
 74     {
 75         j = columns - 1;
 76         do 
 77         {
 78             tableModel.setContentAt(i, j, data.getFormattedValue(i, j));
 79         }
 80         while (j-- > 0)
 81     }
 82     while (i-- > 0)
 83     
 84     tableModel.recalculateMinMaxValues();
 85     
 86     var fisheyeTable = new greg.ross.visualisation.FisheyeTable(tableModel, tablePositionX, tablePositionY, tableWidth, tableHeight, options.tableTitle, this.containerElement);
 87     
 88     fisheyeTable.setBarFill(options.enableBarFill);
 89     fisheyeTable.enableFisheye(options.enableFisheye);
 90 }

 92 greg.ross.visualisation.MagicTable.prototype.escapeHtml = function(text)
 93 {
 94     if (text == null) 
 95         return '';
 96     
 97     return text.replace(/&/g, '&').replace(/</g, '<').replace(/>/g, '>').replace(/"/g, '"');
 98 }
