 21 /**
 22  * This class renders scale cells; i.e. row and column headers.
 23  * @author Greg Ross
 24  * @constructor
 25  * @param tableModel
 26  * @param cellAlignment defines the alignment of text in the cell.
 27  * Must be CellAlignment.LEFT, CellAlignment.CENTRE or CellAlignment.RIGHT;
 28  */
 29 function ScaleCellRenderer(tableModel, cellAlignment)
 30 {
 31 	var defaultCellBackgroundColour = 'rgb(170, 170, 170)';
 32 	var headerOfHeaderBackColour = 'rgb(0, 0, 0)';
 33 	var textColour = 'rgb(0, 0, 0)';
 34 	
 35 	function renderCellText(canvasContext, cellValue, x1, y1, width, height, fontsize)
 36 	{
 37 		if (cellValue == null) return;
 38 		
 39 		var sValue = cellValue + "";
 40 		var font = "sans";
 41 		var h = CanvasTextFunctions.descent(font, fontsize) + fontsize;
 42 		var x = (x1 + width) - width/2;
 43 		var y = (y1 + height) - height/2;
 44 		
 45 		switch (cellAlignment)
 46 		{
 47 			case CellAlignment.LEFT:
 48 			{
 49 				canvasContext.drawText(font, fontsize, x1+5, y + h/2, sValue);
 50 				break;
 51 			}
 52 			case CellAlignment.CENTRE:
 53 			{
 54 				canvasContext.drawTextCenter(font, fontsize, x, y + h/2, sValue);
 55 				break;
 56 			}
 57 			case CellAlignment.RIGHT:
 58 			{
 59 				x = x1 + width - 3;
 60 				canvasContext.drawTextRight(font, fontsize, x, y + h/2, sValue);
 61 				break;
 62 			}
 63 		}
 64 	}
 65 	
 66 	function isHeaderOfHeaders(row, column)
 67 	{
 68 		return (row < tableModel.columnHeaderCount && column < tableModel.rowHeaderCount);
 69 	}
 70 	
 71 	/**
 72 	 * The fisheye table calls this method to render the cell. Classes that extend a cell renderer
 73 	 * must implement this method.
 74 	 * @member ScaleCellRenderer
 75 	 * @param canvasContext the canvas context
 76 	 * @param row the row index
 77 	 * @param column the column index
 78 	 * @param x1 the top-left cell X-coordinate
 79 	 * @param y1 the top-left cell Y-coordinate
 80 	 * @param width the width of the cell
 81 	 * @param height the height of the cell
 82 	 * @param cellValue the value ot be rendered in the cell
 83 	 * @param fisheyeEnabled true if the fisheye feature is turned on
 84 	 * @param barFillEnabled true if the bar-fill feature is turned on
 85 	 */
 86 	this.drawCell = function(canvasContext, row, column, x1, y1, width, height, cellValue)
 87 	{
 88 		var cellIsHeaderOfHeader = isHeaderOfHeaders(row, column);
 89 		
 90 		if (cellIsHeaderOfHeader)
 91 			canvasContext.fillStyle = headerOfHeaderBackColour;
 92 		else 
 93 			canvasContext.fillStyle = defaultCellBackgroundColour;
 94 			
 95 		canvasContext.fillRect(x1, y1, width, height);
 96         canvasContext.lineWidth = 1;
 97 		
 98 		canvasContext.lineWidth = 1.5;
 99 		canvasContext.strokeStyle = 'rgb(220, 220, 220)';
100 		canvasContext.beginPath();
101 		
102 		if (cellIsHeaderOfHeader) 
103 		{
104 			canvasContext.moveTo(x1 + width, y1);
105 			canvasContext.lineTo(x1 + width, y1 + height);
106 			canvasContext.stroke();
107 			canvasContext.strokeStyle = 'rgb(255, 255, 255)';
108 			renderCellText(canvasContext, cellValue, x1, y1, width, height, 12);
109 		}
110 		else 
111 		{
112 			canvasContext.moveTo(x1 + 2, y1 + height);
113 			canvasContext.lineTo(x1 + 2, y1 + 2);
114 			canvasContext.lineTo(x1 + width, y1 + 2);
115 			canvasContext.stroke();
116 			canvasContext.strokeStyle = textColour;
117 			renderCellText(canvasContext, cellValue, x1, y1, width, height, 10);
118 		}
119 	}
120 }
