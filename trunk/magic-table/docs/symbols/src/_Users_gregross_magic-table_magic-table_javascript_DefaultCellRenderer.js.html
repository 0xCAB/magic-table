 21 function DefaultCellRenderer(tableModel, cellAlignment, colourRamp)
 22 {
 23 	var colourGradient;
 24 	var defaultCellBackgroundColour = 'rgb(255, 255, 255)';
 25 	var gridColour = 'rgb(0, 0, 0)';
 26 	
 27 	function init()
 28 	{
 29 		colourGradient = TableGradientColourProvider.createGradient(tableModel, colourRamp);
 30 	}
 31 	
 32 	function renderCellText(canvasContext, cellValue, x1, y1, width, height, fisheyeEnabled)
 33 	{
 34 		if (cellValue == null) return;
 35 		
 36 		var sValue = cellValue + "";
 37 		var font = "sans";
 38 	  	var fontsize = 10;
 39 		
 40 		if (CanvasTextFunctions.measure(font, fontsize, sValue) < width && !fisheyeEnabled) 
 41 		{
 42 			var h = CanvasTextFunctions.descent(font, fontsize) + fontsize;
 43 			var x = (x1 + width) - width / 2;
 44 			var y = (y1 + height) - height / 2;
 45 			
 46 			switch (cellAlignment)
 47 			{
 48 				case CellAlignment.LEFT:
 49 				{
 50 					canvasContext.drawText(font, fontsize, x1+3, y + h/2, sValue);
 51 					break;
 52 				}
 53 				case CellAlignment.CENTRE:
 54 				{
 55 					canvasContext.drawTextCenter(font, fontsize, x, y + h/2, sValue);
 56 					break;
 57 				}
 58 				case CellAlignment.RIGHT:
 59 				{
 60 					x = x1 + width - 3;
 61 					canvasContext.drawTextRight(font, fontsize, x, y + h/2, sValue);
 62 					break;
 63 				}
 64 			}
 65 		}
 66 	}
 67 	
 68 	init();
 69 	
 70 	this.drawCell = function(canvasContext, row, column, x1, y1, width, height, cellValue, fisheyeEnabled, barFillEnabled)
 71 	{
 72 		var colr = TableGradientColourProvider.getColourFromValue(colourGradient, cellValue, defaultCellBackgroundColour);
 73 		
 74 		if (!barFillEnabled) 
 75 		{
 76 			canvasContext.fillStyle = colr;
 77 			canvasContext.fillRect(x1+0.5, y1+0.5, width-0.5, height-0.5);
 78 		}
 79 		else
 80 		{
 81 			canvasContext.fillStyle = defaultCellBackgroundColour;
 82 			canvasContext.fillRect(x1+0.5, y1+0.5, width-0.5, height-0.5);
 83 			
 84 			var barWidth = TableMath.calculateBarWidth(tableModel, cellValue, width, column);
 85 			canvasContext.fillStyle = colr;
 86 			canvasContext.fillRect(x1+0.5, y1+0.5, barWidth, height-0.5);
 87 		}
 88 		
 89 		canvasContext.strokeStyle = gridColour;
 90         canvasContext.lineWidth = 1;
 91 		canvasContext.beginPath();
 92 		canvasContext.moveTo(x1, y1 + height);
 93 		canvasContext.lineTo(x1+width, y1+height);
 94 		canvasContext.lineTo(x1 + width, y1);
 95 		canvasContext.stroke();
 96 		
 97 		renderCellText(canvasContext, cellValue, x1, y1, width, height, fisheyeEnabled);
 98 	}
 99 }
