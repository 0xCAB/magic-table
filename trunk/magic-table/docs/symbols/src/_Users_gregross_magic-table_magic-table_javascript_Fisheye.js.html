 21 /**
 22  * Fisheye lens model
 23  * @constructor
 24  */
 25 greg.ross.visualisation.Fisheye = function()
 26 {
 27 	var focusHeight = 9;
 28 	var lensX = 0;
 29 	var lensY = 0;
 30 	var lensRadius = 30;
 31 
 32 	/**
 33 	 * Return the Euclidean distance of the given point from the lens focus
 34 	 * @member greg.ross.visualisation.Fisheye
 35 	 * @param x
 36 	 * @param y
 37 	 */
 38 	function distanceFromFocus(x, y)
 39 	{
 40 		return distance(x - lensX, y - lensY);
 41 	}
 42 
 43 	/**
 44 	 * Return the height at a point on the lens' surface
 45 	 * @member greg.ross.visualisation.Fisheye
 46 	 * @param distance
 47 	 */
 48 	function heightAtPoint(distance)
 49 	{
 50 		if (focusHeight == 0 || distance > lensRadius)
 51 			return 0;
 52 		else
 53 		{
 54 			var t = distance / lensRadius;
 55 			return Math.min(focusHeight * lens(t), focusHeight);
 56 		}
 57 	}
 58 
 59 	/**
 60 	 * The function that defines the lens shape.
 61 	 * @member greg.ross.visualisation.Fisheye
 62 	 * @param t
 63 	 */
 64 	function lens(t)
 65 	{
 66 		return 1 - t;
 67 	}
 68 
 69 	function getScaleFactor(x, y)
 70 	{
 71 		var height = heightAtPoint(distanceFromFocus(x, y));
 72 		return 10.0 / (10.0 - height);
 73 	}
 74 
 75 	/**
 76 	 * transform the given coordinate according to its position under the lens
 77 	 * @member greg.ross.visualisation.Fisheye
 78 	 * @param point the x coordinate of a point
 79 	 */
 80 	function transformX(x, scale)
 81 	{
 82 		return (x - lensX) * scale + lensX;
 83 	}
 84 
 85 	/**
 86 	 * transform the given coordinate according to its position under the lens
 87 	 * @member greg.ross.visualisation.Fisheye
 88 	 * @param point the y coordinate of a point
 89 	 */
 90 	function transformY(y, scale)
 91 	{
 92 		return (y - lensY) * scale + lensY;
 93 	}
 94 
 95 	/**
 96 	 * transform the given coordinate according to its position under the lens
 97 	 * @member greg.ross.visualisation.Fisheye
 98 	 * @param point the x-y coordinate of a point
 99 	 */
100 	this.transform = function(point)
101 	{
102 		var scaleFactor = getScaleFactor(point.x, point.y);
103 		var newPoint = {x:point.x, y:point.y};
104 
105 		if (scaleFactor != 1)
106 		{
107 			newPoint.x = Math.ceil(transformX(newPoint.x, scaleFactor));
108 			newPoint.y = Math.ceil(transformY(newPoint.y, scaleFactor));
109 			return newPoint;
110 		}
111 
112 		return point;
113 	}
114 	
115 	/**
116 	 * Sets the radius of the lens
117 	 * @member greg.ross.visualisation.Fisheye
118 	 * @param focusRadius
119 	 */
120 	this.setLensRadius = function(focusRadius)
121 	{
122 		lensRadius = focusRadius;
123 	}
124 	
125 	/**
126 	 * Sets the position of the lens.
127 	 * @member greg.ross.visualisation.Fisheye
128 	 * @param x
129 	 * @param y
130 	 */
131 	this.setLensPosition = function(x, y)
132 	{
133 		setLensX(x);
134 		setLensY(y);
135 	}
136 
137 	function setLensX(focusX)
138 	{
139 		if (lensX == focusX)
140 			return;
141 			
142 		lensX = focusX;
143 	}
144 
145 	function setLensY(focusY)
146 	{
147 		if (lensY == focusY) return;
148 		
149 		lensY = focusY;
150 	}
151 	
152 	function distance(dx, dy)
153 	{
154 		return Math.sqrt((dx * dx) + (dy * dy));
155 	}
156 }
