 21 function Fisheye()
 22 {
 23 	var focusHeight = 9;
 24 	var lensX = 0;
 25 	var lensY = 0;
 26 	var lensRadius = 30;
 27 
 28 	function distanceFromFocus(x, y)
 29 	{
 30 		return distance(x - lensX, y - lensY);
 31 	}
 32 
 33 	function heightAtPoint(distance)
 34 	{
 35 		if (focusHeight == 0 || distance > lensRadius)
 36 			return 0;
 37 		else
 38 		{
 39 			var t = distance / lensRadius;
 40 			return Math.min(focusHeight * lens(t), focusHeight);
 41 		}
 42 	}
 43 
 44 	function lens(t)
 45 	{
 46 		return 1 - t;
 47 	}
 48 
 49 	function getScaleFactor(x, y)
 50 	{
 51 		var height = heightAtPoint(distanceFromFocus(x, y));
 52 		return 10.0 / (10.0 - height);
 53 	}
 54 
 55 	function transformX(x, scale)
 56 	{
 57 		return (x - lensX) * scale + lensX;
 58 	}
 59 
 60 	function transformY(y, scale)
 61 	{
 62 		return (y - lensY) * scale + lensY;
 63 	}
 64 
 65 	this.transform = function(point)
 66 	{
 67 		var scaleFactor = getScaleFactor(point.x, point.y);
 68 		var newPoint = {x:point.x, y:point.y};
 69 
 70 		if (scaleFactor != 1)
 71 		{
 72 			newPoint.x = Math.ceil(transformX(newPoint.x, scaleFactor));
 73 			newPoint.y = Math.ceil(transformY(newPoint.y, scaleFactor));
 74 			return newPoint;
 75 		}
 76 
 77 		return point;
 78 	}
 79 	
 80 	this.setLensRadius = function(focusRadius)
 81 	{
 82 		lensRadius = focusRadius;
 83 	}
 84 	
 85 	this.setLensPosition = function(x, y)
 86 	{
 87 		setLensX(x);
 88 		setLensY(y);
 89 	}
 90 
 91 	function setLensX(focusX)
 92 	{
 93 		if (lensX == focusX)
 94 			return;
 95 			
 96 		lensX = focusX;
 97 	}
 98 
 99 	function setLensY(focusY)
100 	{
101 		if (lensY == focusY) return;
102 		
103 		lensY = focusY;
104 	}
105 	
106 	function distance(dx, dy)
107 	{
108 		return Math.sqrt((dx * dx) + (dy * dy));
109 	}
110 }
